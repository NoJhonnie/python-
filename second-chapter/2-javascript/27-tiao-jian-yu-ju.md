#### if else

```
var a = 6;
if(a == 1)
{
    alert('english');
}
else if(a == 1)
{
    alert('mathes');
}
else 
{
    alert('no class');
}
```

#### switch

```
var a = 6;
switch (a){
    case 1:
        alert('语文');
        break;
    case 2:
        alert('数学');
        break;
    case 3:
        alert('英语');
        break;
    default:
        alert('没课');
}
```
